package golang

import "fmt"

func main() {
  s := []int{ 7, 2, 8, -9, 4, 0 }

  c := make(chan int)

  go sum(s[:len(s) / 2], c)
  go sum(s[len(s) / 2:], c)
  
  x, y := <- c, <- c // receive from c

  fmt.Println(x, y, x + y)
}
