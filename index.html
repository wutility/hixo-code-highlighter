Syntax Highlighter

  • Extensible.
  • Simple.
  • Lightweight.
  • Fast compared with others syntax highlighters.
  • Written in Javascript.
  • Zero dependencies.
package golang

import "fmt"

func main() {
  s := []int{ 7, 2, 8, -9, 4, 0 }

  c := make(chan int)

  go sum(s[:len(s) / 2], c)
  go sum(s[len(s) / 2:], c)
  
  x, y := <- c, <- c // receive from c

  fmt.Println(x, y, x + y)
}

# Usage

import Hixo from 'hixo';
import 'hixo/build/hixo.css';

const options = { language: 'java', lineNum: true }

const hixo = new Hixo(options); 
const preElement = document.querySelector('pre')
const codeStr = 'System.out.println("Hello");';

preElement.innerHTML = hixo.codeToHtml(codeStr)

# Methods

// code to html string
hixo.codeToHtml('System.out.println("Hello");')

// add new keywords
hixo.addKeys('final|struct|range|async');

// add new regular expression
hixo.addRegex({ 
  pattern: /(?![.])[:$]{0,2}(\w+)(?=\(.)/g, // required
  color: 'method', // required
  italic: true, // optional
  bold: true, // optional
  stripHtml: true // optional: remove any html tag
});

// highlight all pre elements
// Html: <pre data-language="javascript"></pre>
hixo.highlightAll()
